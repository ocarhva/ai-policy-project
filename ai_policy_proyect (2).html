bias_keywords = [
    "different",
    "uncomfortable",
    "ignored",
    "rude",
    "disrespectful",
    "poor treatment"
]

def detect_bias(text):
    text = text.lower()
    return any(keyword in text for keyword in bias_keywords)

reviews["potential_bias_signal"] = reviews["review"].apply(detect_bias)
